CREATE OR REPLACE TABLE
`engine-analytics-490008.engine_health_data.engine_summary` AS

SELECT
  engine_id,
  MAX(cycle) AS total_cycles,
  AVG(sensor_4) AS avg_sensor_4,
  AVG(sensor_7) AS avg_sensor_7,
  AVG(sensor_11) AS avg_sensor_11,
  AVG(sensor_15) AS avg_sensor_15
FROM `engine-analytics-490008.engine_health_data.engine_rul`
GROUP BY engine_id;
